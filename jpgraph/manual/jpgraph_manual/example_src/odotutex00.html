<?php // content="text/plain; charset=utf-8"
require_once ("jpgraph/jpgraph.php");
require_once ("jpgraph/jpgraph_odo.php");
 
// 新規オドメータ・グラフ・エリアを作成。(幅250, 高さ200ピクセル)
$graph = new OdoGraph(250,140);
 
// オドメータを作成しグラフ・エリアに追加。
// 初期設定によりスケール（目盛り）は0から100。
$odo = new Odometer();
 
// オドメータの指針が指す値を設定。
$odo->needle->Set(30);
 
// オドメータをグラフ・エリアに追加。
$graph->Add($odo);
 
// ... 最後に、完成した画像をブラウザに返す。
$graph->Stroke();
 
?>
