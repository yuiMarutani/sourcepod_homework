<?php // content="text/plain; charset=utf-8"
 
require_once '../jpgraph.php';
require_once '../jpgraph_contourf.php';
 
// 等高線グラフを作成するためのデータを指定する
$data = array(
    array (12,12,10,10),
    array (10,10,8,14),
    array (7,7,13,17),
    array (4,5,8,12),
    array (10,8,7,8));
 
// 基本的なグラフ領域を指定する
$graph = new Graph(300,300);
$graph->SetMargin(30, 30, 40, 30);
$graph->SetScale('intint');
$graph->SetMarginColor('white');
 
// グラフのタイトルを設定する
$graph->title->Set('Filled contour plot');
$graph->title->SetFont(FF_VERDANA,FS_BOLD,12);
 
$graph->subtitle->Set('(No lines, no labels)');
$graph->subtitle->SetFont(FF_VERDANA,FS_ITALIC,10);
 
// 新しい塗りつぶし等高線プロットを作成する
$cp = new FilledContourPlot($data);
 
// 赤色と青色のみのスキーマを使用する
$cp->UseHighContrastColor(true);
 
// 視覚的に反転させる
$cp->SetInvert();
 
// 等高線グラフに色を塗る
$cp->SetFilled(true);
 
// ラベル無し
$cp->ShowLabels(false);
 
// 線無し
$cp->ShowLines(false);
 
// プロットをグラフに追加する
$graph->Add($cp);
 
// クライアントに結果を戻す
$graph->stroke();
 
?>
